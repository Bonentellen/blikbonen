blikbonen.importers.coinbase

An importer used for Coinbase activity .csv files.

Example

The following example configures this importer:

ACCOUNT = "Assets:Coinbase"
return coinbase.Importer(
    {
        "account_number": "897414yb-147214yu-87243b5r-3248yb",
        "main_account": f"{ACCOUNT}:{{ticker}}",
        "cash_account": f"{ACCOUNT}:{{currency}}",
        "transfer": "Assets:Zero-Sum-Accounts:Coinbase",
        "learning_reward": "Income:CoinbaseEarn",
        "staking": "Income:CoinbaseRewards",
        "cg": "Income:CapitalGains",
        "fees": "Expenses:TradingFees",
    }
)

 37def _advanced_regex(a: str):
 38    return re.match(
 39        r"^(Bought|Sold) (\d+\.?\d*) ([A-Z0-9]+) for (\d+\.?\d*) ([A-Z0-9]+) on ([A-Z0-9]+)-([A-Z0-9]+) at (\d+\.?\d*) ([A-Z0-9]+\/[A-Z0-9]+)$",
 40        a,
 41    )
 42
 43
 44def _advanced_buy_sell_add_amount(a) -> str | None:
 45    m = _advanced_regex(a["Notes"])
 46    if m is not None:
 47        if m.group(1) == "Bought":
 48            return m.group(2)
 49        return m.group(4)
 50    return None
 51
 52
 53def _advanced_buy_sell_subtract_amount(a) -> str | None:
 54    m = _advanced_regex(a["Notes"])
 55    if m is not None:
 56        if m.group(1) == "Bought":
 57            return m.group(4)
 58        return m.group(2)
 59    return None
 60
 61
 62def _advanced_buy_sell_add_currency(a) -> str | None:
 63    m = _advanced_regex(a["Notes"])
 64    if m is not None:
 65        if m.group(1) == "Bought":
 66            return m.group(3)
 67        return m.group(5)
 68    return None
 69
 70
 71def _advanced_buy_sell_subtract_currency(a) -> str | None:
 72    m = _advanced_regex(a["Notes"])
 73    if m is not None:
 74        if m.group(1) == "Bought":
 75            return m.group(5)
 76        return m.group(3)
 77    return None
 78
 79
 80def _advanced_buy_sell_market(a) -> str | None:
 81    m = _advanced_regex(a["Notes"])
 82    if m is not None:
 83        return m.group(9)
 84    return None

140    @override
141    def prepare_table(self, rdr: etl.Table):
142        rdr = etl.addfield(rdr, "tradeDate", lambda r: r["Timestamp"])
143
144        # Parse the advanced trade options
145        rdr = etl.addfield(rdr, "advanced_add_amount", _advanced_buy_sell_add_amount)
146        rdr = etl.addfield(rdr, "advanced_add_currency", _advanced_buy_sell_add_currency)
147        rdr = etl.addfield(rdr, "advanced_subtract_amount", _advanced_buy_sell_subtract_amount)
148        rdr = etl.addfield(rdr, "advanced_subtract_currency", _advanced_buy_sell_subtract_currency)
149        rdr = etl.addfield(rdr, "advanced_market", _advanced_buy_sell_market)
150
151        # Some advanced trade options can be interpreted as normal buy/sell
152        rdr = etl.convert(
153            rdr,
154            "Transaction Type",
155            lambda _r: "sellstock",
156            where=lambda r: r["advanced_add_currency"] == self.currency,
157        )
158        rdr = etl.convert(
159            rdr,
160            "Transaction Type",
161            lambda _r: "buystock",
162            where=lambda r: r["advanced_subtract_currency"] == self.currency,
163        )
164        # The remaining entries have two different currencies, neither `self.currency`
165
166        # Fix sell postings: if `units` is negative, `total` will be used for
167        # the other posting and should be positive
168        return etl.convert(
169            rdr,
170            "Total (inclusive of fees and/or spread)",
171            lambda v: v[1:],
172            where=lambda r: (
173                r["Total (inclusive of fees and/or spread)"].startswith("-")
174                and r["Quantity Transacted"].startswith("-")
175            ),
176        )
177

181        self.config["capgainsd_st"] = "Income:UNUSED"
182        self.config["dividends"] = "Income:UNUSED"
183        self.config["dividends"] = "Income:UNUSED"
184        self.config["interest"] = "Income:UNUSED"
185        self.config["fund_info"] = {"fund_data": [], "money_market": []}
186        self.currency = "EUR"
187        self.max_rounding_error = 0.0000000000000001
188        self.filename_pattern_def = f"{self.config['account_number']}_.*\\.csv"
189        self.column_labels_line = "ID,Timestamp,Transaction Type,Asset,Quantity Transacted,Price Currency,Price at Transaction,Subtotal,Total (inclusive of fees and/or spread),Fees and/or Spread,Notes"
190        self.header_identifier = ""
191        self.date_format = "%Y-%m-%d %H:%M:%S UTC"
192        self.header_map = {

216    def _generate_advanced_trade_entry(self, ot: etl.Record, file: str, counter):
217        metadata = data.new_metadata(file, next(counter))
218        metadata.update(self.build_metadata(file, metatype="transaction_transfer", data={"transaction": ot}))
219        ticker_add, _ticker_add_long_name = self.get_ticker_info(ot.advanced_add_currency)
220        ticker_sub, _ticker_sub_long_name = self.get_ticker_info(ot.advanced_subtract_currency)
221        units_add = decimal.Decimal(ot.advanced_add_amount)
222        units_sub = -abs(decimal.Decimal(ot.advanced_subtract_amount))
223        add_account = self.get_acct("main_account", ot, ticker_add)
224        sub_account = self.get_acct("main_account", ot, ticker_sub)
225        narration = ot.advanced_market
226        date = ot.date.date()
227        payee = self.get_payee(ot)
228
229        entry = data.Transaction(
230            metadata,
231            date,
232            self.FLAG,
233            payee,
234            narration,
235            self.get_tags(ot),
236            self.get_links(ot),
237            [],
238        )
239        # Since the price is given in CURR1/CURR2, we need to compute it manually
240        unit_sub_price = abs(ot.amount / units_sub)
241        unit_add_price = abs(ot.total / units_add)
242        common.create_simple_posting_with_cost_or_price(
243            entry,
244            sub_account,
245            units_sub,
246            ticker_sub,
247            price_number=unit_sub_price,
248            price_currency=self.currency,
249            costspec=CostSpec(None, None, None, None, None, None),
250            price_cost_both_zero_handler=self.price_cost_both_zero_handler,
251            ot=ot,
252        )
253        # Profit/Loss
254        cg_acct = self.get_acct("cg", ot, ticker_sub)
255        data.create_simple_posting(entry, cg_acct, None, None)
256        common.create_simple_posting_with_cost(
257            entry,
258            add_account,
259            units_add,
260            ticker_add,
261            unit_add_price,
262            self.currency,
263            self.price_cost_both_zero_handler,
264            ot=ot,
265        )
266        return entry

268    def _generate_staking_entry(self, ot, file, counter):
269        metadata = data.new_metadata(file, next(counter))
270        metadata.update(self.build_metadata(file, metatype="transaction_transfer", data={"transaction": ot}))
271        ticker, _ticker_long_name = self.get_ticker_info(ot.security)
272        units = ot.units
273        main_acct = self.get_acct("main_account", ot, ticker)
274        narration = self.security_narration(ot)
275        date = ot.date.date()
276
277        payee = self.get_payee(ot)
278        if ot.type == "staking":
279            payee = "Staking Income"
280
281        entry = data.Transaction(
282            metadata,
283            date,
284            self.FLAG,
285            payee,
286            narration,
287            self.get_tags(ot),
288            self.get_links(ot),
289            [],
290        )
291
292        if ot.type == "staking":
293            target_account = self.config["staking"]
294        elif ot.type == "learning_reward":
295            target_account = self.config["learning_reward"]
296        else:
297            msg = f"transaction type {ot.type} unknown"
298            raise ValueError(msg)
299
300        common.create_simple_posting_with_cost(entry, main_acct, units, ticker, ot.unit_price, ot.currency, ot=ot)
301        common.create_simple_posting_with_cost(
302            entry, target_account, -1 * units, ticker, ot.unit_price, ot.currency, ot=ot
303        )
304
305        return entry

307    @override
308    def extract_transactions(self, file: str, counter):
309        new_entries = []
310        self.read_file(file)
311        for ot in self.get_transactions():
312            if self.skip_transaction(ot):
313                continue
314            if ot.type in [
315                "buymf",
316                "sellmf",
317                "buystock",
318                "buydebt",
319                "sellstock",
320                "buyother",
321                "sellother",
322                "reinvest",
323            ]:
324                entry = self.generate_trade_entry(ot, file, counter)
325            elif ot.type in [
326                "other",
327                "credit",
328                "debit",
329                "transfer",
330                "xfer",
331                "dep",
332                "income",
333                "fee",
334                "dividends",
335                "capgainsd_st",
336                "capgainsd_lt",
337                "cash",
338                "payment",
339                "check",
340                "invexpense",
341            ]:
342                entry = self.generate_transfer_entry(ot, file, counter)
343            elif ot.type in ["staking", "learning_reward"]:
344                entry = self._generate_staking_entry(ot, file, counter)
345            elif ot.type == "advanced":
346                entry = self._generate_advanced_trade_entry(ot, file, counter)
347            else:
348                msg = f"unknown entry type {ot.type}"
349                raise ValueError(msg)
350            self.add_fee_postings(entry, ot)
351            self.add_custom_postings(entry, ot)
352            new_entries.append(entry)
353        return new_entries

REQUIRED_CONFIG = { 'account_number': 'account number', 'main_account' : 'Destination account of import', 'cash_account' : 'Cash account (usually same as the main account + a :{currency} appended)', 'transfer' : 'Account to which contributions and outgoing is transferred', # transfer account is optional. If left off no target posting will be created. # This allows for additional tools to handle this like smart importer. 'dividends' : 'Account to book dividends', 'cg' : 'Account to book capital gains/losses', 'capgainsd_lt' : 'Account to book long term capital gains distributions to' 'capgainsd_st' : 'Account to book short term capital gains distributions to' 'fees' : 'Account to book fees to', 'invexpense' : 'Account to book expenses (like foreign taxes) to', 'rounding_error': 'Account to book rounding errors to', 'fund_info ' : 'dictionary of fund info (by_id, money_market)', }

Certian variables in the config are resolved by this transaction builder. Those variables are:

{ticker}: replaced with the ticker symbol. Use this to obtain account names that end with the commodity (commodity leaf accounts).

{currency}: replaced with the operating currency of the account in question (which is in turn obtained from the ofx or csv)

{source401k}: replaced with the source of the 401(k) funding, like 'Pretax' or 'Match' (for employer matching). Used only in 401(k) accounts. Practically needed only in 'main_account'.

Example: { 'account_number' : '1234567', 'main_account' : 'Assets:Investments:XTrade:{ticker}', 'cash_account' : 'Assets:Investments:XTrade:{currency}', 'transfer' : 'Assets:Zero-Sum-Accounts:Transfers:Bank-Account', 'dividends' : 'Income:Dividends:XTrade:{ticker}', 'interest' : 'Income:Interest:XTrade:{ticker}', 'cg' : 'Income:Capital-Gains:XTrade:{ticker}', 'capgainsd_lt' : 'Income:Capital-Gains-Distributions:Long:XTrade:{ticker}', 'capgainsd_st' : 'Income:Capital-Gains-Distributions:Short:XTrade:{ticker}', 'fees' : 'Expenses:Brokerage-Fees:XTrade', 'invexpense' : 'Expenses:Investment-Expenses:XTrade', 'rounding_error' : 'Equity:Rounding-Errors:Imports', 'fund_info' : fund_info, }

For a 401(k) account you could set:

'main_account'   : 'Assets:Vanguard:401k:{source401k}:{ticker}',
