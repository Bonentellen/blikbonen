blikbonen.importers.degiro.account

A DeGiro account .csv Importer. This imports transactions, including buys and sells, but cannot generate balance statements. For this, use the blikbonen.importers.degiro.portfolio importer.

Example

The following example shows how you can initialize this importer:

from blikbonen.importers.degiro import account

ACCOUNT = "Assets:DeGiro"
FUND_INFO = {
    "fund_data": [
        (
            "VUSD",
            "IE00B3XXRP09",
            "Vanguard S&P 500 UCITS ETF (USD) Distributing",
        ),
    ],
    "money_market": [],
}
importer = account.Importer(
    {
        "account_number": "NL12345678901234567890",
        "main_account": f"{ACCOUNT}:{{ticker}}",
        "cash_account": f"{ACCOUNT}:{{currency}}",
        "transfer": "Assets:Zero-Sum-Accounts:DeGiro",
        "interest": "Income:Interest",
        "dividends": "Income:Dividend:{ticker}",
        "cg": "Income:CG",
        "rounding_error": "Equity:RoundingError",
        "tax": "Expenses:Tax",
        "service_fee": "Expenses:ServiceFee",
        "fund_info": FUND_INFO,
        "fees": "Expenses:TradingFees",
    }
)

161    def _get_units(self, row) -> str | None:
162        """Get the units for this transaction. We fetch it from the description
163        in case of a buy/sell or use the amount in case of a currency
164        conversion.
165        """
166        m = re.match(_BUY_REGEX, row["Omschrijving"])
167        if m is not None:
168            return m.group(1)
169        m = re.match(_SELL_REGEX, row["Omschrijving"])
170        if m is not None:
171            return "-" + m.group(1)
172        if row["Omschrijving"] in [_CURRENCY_CREDIT, _CURRENCY_DEBET] and row["Mutatie"] != self.currency:
173            return row["mutatie_units"]
174        return None
175
176    def _get_security(self, row) -> str | None:
177        if row["Omschrijving"] in [_CURRENCY_CREDIT, _CURRENCY_DEBET] and row["Mutatie"] != self.currency:
178            return row["Mutatie"]
179        return row["ISIN"]
180
181    def _get_field_type(self, row) -> str:
182        if (
183            row["Omschrijving"].startswith("Overboeking van uw geldrekening bij flatexDEGIRO Bank")
184            or row["Omschrijving"].startswith("Overboeking naar uw geldrekening bij flatexDEGIRO Bank")
185            or row["Omschrijving"] == "SEPA Instant Terugstorting"
186            or row["Omschrijving"] == "flatex Storting"
187            or row["Omschrijving"] == "iDEAL Deposit"
188            or row["Omschrijving"] == "flatex terugstorting"
189        ):
190            return "cash"
191        if row["Omschrijving"].startswith("Koop "):
192            return "buystock"
193        if row["Omschrijving"].startswith("Verkoop "):
194            return "sellstock"
195        if row["Omschrijving"] in ["B.T.W.", "Service-fee", "Dividendbelasting"]:
196            return "fee"
197        if row["Omschrijving"].startswith("DEGIRO Aansluitingskosten"):
198            return "fee"
199        if row["Omschrijving"] == "Flatex Interest Income":
200            return "income"
201        if row["Omschrijving"] == "Dividend":
202            return "dividends"
203        if row["Omschrijving"] in [_CURRENCY_CREDIT, _CURRENCY_DEBET] and row["Mutatie"] == self.currency:
204            return "cash"
205        if row["Omschrijving"] == _CURRENCY_CREDIT and row["Mutatie"] != self.currency:
206            return "buyother"
207        if row["Omschrijving"] == _CURRENCY_DEBET and row["Mutatie"] != self.currency:
208            return "sellother"
209        if row["Omschrijving"] == "Terugstorting in behandeling":
210            return "cash"
211        msg = f"transaction type not determined: {row['Omschrijving']}"
212        raise ValueError(msg)
213
214    def _get_amount(self, cur):
215        m = re.match(_TRANSFER_FROM_REGEX, cur["Omschrijving"])
216        if m is not None:
217            return f"-{m.group(1)}".replace(".", "").replace(",", ".")
218        m = re.match(_TRANSFER_TO_REGEX, cur["Omschrijving"])
219        if m is not None:
220            return f"{m.group(1)}".replace(".", "").replace(",", ".")
221        return cur["mutatie_units"]

232    def _get_total_from_context(self, prv, row, nxt):
233        if row["Omschrijving"].startswith("Koop ") or row["Omschrijving"].startswith("Verkoop "):
234            return row["mutatie_units"]
235        # For currency conversions, there is one debit and one credit
236        # transaction. We use the other to get the total
237        if (
238            nxt is not None
239            and row["Omschrijving"] == _CURRENCY_DEBET
240            and nxt["Omschrijving"] == _CURRENCY_CREDIT
241            and row["Mutatie"] != self.currency
242            and nxt["Mutatie"] == self.currency
243        ):
244            return nxt["mutatie_units"]
245        if (
246            prv is not None
247            and row["Omschrijving"] == _CURRENCY_CREDIT
248            and prv["Omschrijving"] == _CURRENCY_DEBET
249            and row["Mutatie"] != self.currency
250            and prv["Mutatie"] == self.currency
251        ):
252            return prv["mutatie_units"]
253        return row["amount"]
254
255    def _get_unit_price(self, row):
256        m = re.match(_BUY_REGEX, row["Omschrijving"])
257        if m is not None:
258            return m.group(2).replace(",", ".")
259        m = re.match(_SELL_REGEX, row["Omschrijving"])
260        if m is not None:
261            return m.group(2).replace(",", ".")
262        if row["Omschrijving"] in [_CURRENCY_CREDIT, _CURRENCY_DEBET] and row["Mutatie"] != self.currency:
263            return 1 / decimal.Decimal(row["FX"].replace(",", "."))
264        return None
265
266    def _get_currency(self, row):
267        for reg in [_BUY_REGEX, _SELL_REGEX]:
268            m = re.match(reg, row["Omschrijving"])
269            if m is not None:
270                return m.group(3)
271        for reg in [_TRANSFER_FROM_REGEX, _TRANSFER_TO_REGEX]:
272            m = re.match(reg, row["Omschrijving"])
273            if m is not None:
274                return m.group(2)
275        return row["Mutatie"]

277    def _remove_useless_debit_credit(self, row):
278        """
279        We have moved these postings to the other row, so we can remove them.
280        """
281        return not (row["Omschrijving"] in [_CURRENCY_CREDIT, _CURRENCY_DEBET] and row["Mutatie"] == self.currency)

283    @override
284    def prepare_table(self, rdr):
285        # Ignore these internal transactions
286        rdr = etl.selectne(rdr, "Omschrijving", "Degiro Cash Sweep Transfer")
287        rdr = etl.selectne(rdr, "Omschrijving", "Processed Flatex Withdrawal")
288        rdr = etl.selectne(rdr, "Omschrijving", "Reservation iDEAL")
289        rdr = etl.select(
290            rdr,
291            lambda r: (not r["Omschrijving"].startswith("Overboeking van uw geldrekening bij flatexDEGIRO Bank ")),
292        )
293
294        # Reverse rows (keeping header first)
295        header = etl.header(rdr)
296        rows = list(etl.data(rdr))
297        rdr = etl.wrap([header, *list(reversed(rows))])
298
299        # Add names to unnamed columns
300        rdr = etl.rename(rdr, 8, "mutatie_units")
301        rdr = etl.rename(rdr, 10, "saldo_units")
302
303        rdr = etl.convert(rdr, "mutatie_units", lambda v: v.replace(",", "."))
304        rdr = etl.convert(rdr, "saldo_units", lambda v: v.replace(",", "."))
305
306        # Add the fee to the previous transaction
307        rdr = etl.addfieldusingcontext(rdr, "fees", _get_fee_field_from_context)
308        rdr = etl.selectusingcontext(rdr, _remove_fee_transaction_from_context)
309
310        rdr = etl.addfield(rdr, "type", self._get_field_type)
311        rdr = etl.addfield(rdr, "units", self._get_units)
312        rdr = etl.addfield(rdr, "amount", self._)
313        rdr = etl.addfield(rdr, "security", self._get_security)
314        rdr = etl.addfieldusingcontext(rdr, "total", self._get_total_from_context)
315        rdr = etl.addfield(rdr, "unit_price", self._get_unit_price)
316        rdr = etl.addfield(rdr, "currency", self._get_currency)
317        return etl.select(rdr, self._remove_useless_debit_credit)

357    def _fix_currency_for_buy_sell(self, entries: list[data.Directive]):
358        """The currency for buy and sell transactions (not for currency
359        conversions) has been set to `self.currency` at this point, even if
360        there is an intermediate currency involved. We fix this here.
361        """
362        new_entries = []
363        for entry in entries:
364            if isinstance(entry, data.Transaction):
365                # Get the correct currency from metadata and remove
366                expected_currency = entry.meta["currency"]
367                del entry.meta["currency"]
368
369                # We find the transaction cost posting, so we can skip both
370                # legs later on.
371                transaction_cost_posting = None
372                for posting in entry.postings:
373                    if posting.account == self.config["fees"]:
374                        transaction_cost_posting = posting
375                        assert transaction_cost_posting.units is not None  # noqa: S101
376                skipped_transaction_cost_other_leg = False
377
378                if entry.postings[0].units is not None and entry.postings[0].units.currency != expected_currency:
379                    for i in range(len(entry.postings)):
380                        posting = entry.postings[i]
381
382                        # Skip the transaction cost posting, as well as the opposit leg
383                        if transaction_cost_posting is not None:
384                            assert transaction_cost_posting.units is not None  # noqa: S101
385                            # Always skip the transaction cost posting (there should only be 1)
386                            if posting == transaction_cost_posting:
387                                continue
388                            # Since it could be that other legs have the same units, only skip this once.
389                            if (
390                                posting.units == -transaction_cost_posting.units
391                                and not skipped_transaction_cost_other_leg
392                            ):
393                                skipped_transaction_cost_other_leg = True
394                                continue
395
396                        if posting.units is not None and posting.units.currency == self.currency:
397                            posting = posting._replace(units=posting.units._replace(currency=expected_currency))
398                            entry.postings[i] = posting
399                        if posting.price is not None and posting.price.currency != expected_currency:
400                            posting = posting._replace(price=posting.price._replace(currency=expected_currency))
401                            entry.postings[i] = posting
402                        if posting.cost is not None and posting.cost.currency != expected_currency:
403                            posting = posting._replace(cost=posting.cost._replace(currency=expected_currency))
404                            entry.postings[i] = posting
405                        posting = posting._replace(account=posting.account.replace(self.currency, expected_currency))
406                        entry.postings[i] = posting
407
408            new_entries += [entry]
409        return new_entries

411    @override
412    def custom_entry_mods(self, new_entries):
413        new_entries = switch_payee_narration(new_entries)
414
415        # There is one thing we should fix: for some buy/sell statements the
416        # currency is incorrectly set to `self.currency`. It should be
417        # `currency` instead. We can recognize these postings by the fact that
418        # the first posting does not have this as a currency. (None of the
419        # postings do, in fact.)
420        #
421        # A better solution would be to fix this in the importer.
422        new_entries = self._fix_currency_for_buy_sell(new_entries)
423
424        # At this point, the transactions are well-formed. Now we will merge
425        # transactions with the same time & date.
426        old_entries = new_entries
427        new_entries = []
428        for entry in old_entries:
429            if (
430                len(new_entries) > 0
431                and new_entries[-1].date == entry.date
432                and new_entries[-1].meta["time"] == entry.meta["time"]
433            ):
434                for posting in entry.postings:
435                    existing_posting_pos = None
436                    for i in range(len(new_entries[-1].postings)):
437                        existing_posting = new_entries[-1].postings[i]
438                        if posting.account == existing_posting.account:
439                            if posting.units is None and existing_posting.units is None:
440                                existing_posting_pos = i
441                                break
442                            if (
443                                posting.units.currency == existing_posting.units.currency
444                                and posting.price == existing_posting.price
445                                and posting.cost == existing_posting.cost
446                            ):
447                                existing_posting_pos = i
448                                break
449
450                    if existing_posting_pos is None:
451                        new_entries[-1].postings.append(posting)
452                    # We can merge with an existing posting
453                    elif new_entries[-1].postings[existing_posting_pos].units is not None:
454                        new_entries[-1].postings[existing_posting_pos] = (
455                            new_entries[-1]
456                            .postings[existing_posting_pos]
457                            ._replace(
458                                units=new_entries[-1]
459                                .postings[existing_posting_pos]
460                                .units._replace(
461                                    number=new_entries[-1].postings[existing_posting_pos].units.number
462                                    + posting.units.number
463                                )
464                            )
465                        )
466            else:
467                new_entries.append(entry)
468
469        # We simplify transactions as follows: if there is a cost not
470        # in `self.currency`, we convert it by using the cost of this currency
471        # in terms of `self.currency`. We can then remove the adding and
472        # subtracting of this other currency.
473        old_entries = new_entries
474        new_entries = []
475        for entry in old_entries:
476            costs = [[posting.units.currency, posting.cost] for posting in entry.postings if posting.cost is not None]
477
478            for i in range(len(entry.postings)):
479                posting = entry.postings[i]
480                if posting.cost is not None and posting.cost.currency != self.currency:
481                    correct_posting = None
482                    for from_cur, cost in costs:
483                        if from_cur == posting.cost.currency:
484                            correct_posting = cost
485                    if correct_posting is None:
486                        continue
487                    entry.postings[i] = posting._replace(
488                        cost=posting.cost._replace(
489                            currency=self.currency,
490                            number=(posting.cost.number * correct_posting.number).quantize(decimal.Decimal("0.0001")),
491                        )
492                    )
493
494            new_entries.append(entry)
495
496        # We simplify one last time, to filter out combinations like 100 USD {0.8 EUR} & -100 USD {}
497        old_entries = new_entries
498        new_entries = []
499        for entry in old_entries:
500            cancelling_duo = None
501            for i in range(len(entry.postings)):
502                for j in range(i + 1, len(entry.postings)):
503                    if (
504                        entry.postings[i].account == entry.postings[j].account
505                        and entry.postings[i].units is not None
506                        and entry.postings[j].units is not None
507                        and entry.postings[i].units.currency == entry.postings[j].units.currency
508                        and entry.postings[i].units.number == -entry.postings[j].units.number
509                    ):
510                        cancelling_duo = [i, j]
511
512            if cancelling_duo is not None:
513                i, j = cancelling_duo
514                entry.postings.pop(j)
515                entry.postings.pop(i)
516
517            new_entries.append(entry)
518
519        return new_entries

358    def _fix_currency_for_buy_sell(self, entries: list[data.Directive]):
359        """The currency for buy and sell transactions (not for currency
360        conversions) has been set to `self.currency` at this point, even if
361        there is an intermediate currency involved. We fix this here.
362        """
363        new_entries = []
364        for entry in entries:
365            if isinstance(entry, data.Transaction):
366                # Get the correct currency from metadata and remove
367                expected_currency = entry.meta["currency"]
368                del entry.meta["currency"]
369
370                # We find the transaction cost posting, so we can skip both
371                # legs later on.
372                transaction_cost_posting = None
373                for posting in entry.postings:
374                    if posting.account == self.config["fees"]:
375                        transaction_cost_posting = posting
376                        assert transaction_cost_posting.units is not None  # noqa: S101
377                skipped_transaction_cost_other_leg = False
378
379                if entry.postings[0].units is not None and entry.postings[0].units.currency != expected_currency:
380                    for i in range(len(entry.postings)):
381                        posting = entry.postings[i]
382
383                        # Skip the transaction cost posting, as well as the opposit leg
384                        if transaction_cost_posting is not None:
385                            assert transaction_cost_posting.units is not None  # noqa: S101
386                            # Always skip the transaction cost posting (there should only be 1)
387                            if posting == transaction_cost_posting:
388                                continue
389                            # Since it could be that other legs have the same units, only skip this once.
390                            if (
391                                posting.units == -transaction_cost_posting.units
392                                and not skipped_transaction_cost_other_leg
393                            ):
394                                skipped_transaction_cost_other_leg = True
395                                continue
396
397                        if posting.units is not None and posting.units.currency == self.currency:
398                            posting = posting._replace(units=posting.units._replace(currency=expected_currency))
399                            entry.postings[i] = posting
400                        if posting.price is not None and posting.price.currency != expected_currency:
401                            posting = posting._replace(price=posting.price._replace(currency=expected_currency))
402                            entry.postings[i] = posting
403                        if posting.cost is not None and posting.cost.currency != expected_currency:
404                            posting = posting._replace(cost=posting.cost._replace(currency=expected_currency))
405                            entry.postings[i] = posting
406                        posting = posting._replace(account=posting.account.replace(self.currency, expected_currency))
407                        entry.postings[i] = posting
408
409            new_entries += [entry]
410        return new_entries

413    def custom_entry_mods(self, new_entries):
414        new_entries = switch_payee_narration(new_entries)
415
416        # There is one thing we should fix: for some buy/sell statements the
417        # currency is incorrectly set to `self.currency`. It should be
418        # `currency` instead. We can recognize these postings by the fact that
419        # the first posting does not have this as a currency. (None of the
420        # postings do, in fact.)
421        #
422        # A better solution would be to fix this in the importer.
423        new_entries = self._fix_currency_for_buy_sell(new_entries)
424
425        # At this point, the transactions are well-formed. Now we will merge
426        # transactions with the same time & date.
427        old_entries = new_entries
428        new_entries = []
429        for entry in old_entries:
430            if (
431                len(new_entries) > 0
432                and new_entries[-1].date == entry.date
433                and new_entries[-1].meta["time"] == entry.meta["time"]
434            ):
435                for posting in entry.postings:
436                    existing_posting_pos = None
437                    for i in range(len(new_entries[-1].postings)):
438                        existing_posting = new_entries[-1].postings[i]
439                        if posting.account == existing_posting.account:
440                            if posting.units is None and existing_posting.units is None:
441                                existing_posting_pos = i
442                                break
443                            if (
444                                posting.units.currency == existing_posting.units.currency
445                                and posting.price == existing_posting.price
446                                and posting.cost == existing_posting.cost
447                            ):
448                                existing_posting_pos = i
449                                break
450
451                    if existing_posting_pos is None:
452                        new_entries[-1].postings.append(posting)
453                    # We can merge with an existing posting
454                    elif new_entries[-1].postings[existing_posting_pos].units is not None:
455                        new_entries[-1].postings[existing_posting_pos] = (
456                            new_entries[-1]
457                            .postings[existing_posting_pos]
458                            ._replace(
459                                units=new_entries[-1]
460                                .postings[existing_posting_pos]
461                                .units._replace(
462                                    number=new_entries[-1].postings[existing_posting_pos].units.number
463                                    + posting.units.number
464                                )
465                            )
466                        )
467            else:
468                new_entries.append(entry)
469
470        # We simplify transactions as follows: if there is a cost not
471        # in `self.currency`, we convert it by using the cost of this currency
472        # in terms of `self.currency`. We can then remove the adding and
473        # subtracting of this other currency.
474        old_entries = new_entries
475        new_entries = []
476        for entry in old_entries:
477            costs = [[posting.units.currency, posting.cost] for posting in entry.postings if posting.cost is not None]
478
479            for i in range(len(entry.postings)):
480                posting = entry.postings[i]
481                if posting.cost is not None and posting.cost.currency != self.currency:
482                    correct_posting = None
483                    for from_cur, cost in costs:
484                        if from_cur == posting.cost.currency:
485                            correct_posting = cost
486                    if correct_posting is None:
487                        continue
488                    entry.postings[i] = posting._replace(
489                        cost=posting.cost._replace(
490                            currency=self.currency,
491                            number=(posting.cost.number * correct_posting.number).quantize(decimal.Decimal("0.0001")),
492                        )
493                    )
494
495            new_entries.append(entry)
496
497        # We simplify one last time, to filter out combinations like 100 USD {0.8 EUR} & -100 USD {}
498        old_entries = new_entries
499        new_entries = []
500        for entry in old_entries:
501            cancelling_duo = None
502            for i in range(len(entry.postings)):
503                for j in range(i + 1, len(entry.postings)):
504                    if (
505                        entry.postings[i].account == entry.postings[j].account
506                        and entry.postings[i].units is not None
507                        and entry.postings[j].units is not None
508                        and entry.postings[i].units.currency == entry.postings[j].units.currency
509                        and entry.postings[i].units.number == -entry.postings[j].units.number
510                    ):
511                        cancelling_duo = [i, j]
512
513            if cancelling_duo is not None:
514                i, j = cancelling_duo
515                entry.postings.pop(j)
516                entry.postings.pop(i)
517
518            new_entries.append(entry)
519
520        return new_entries

416        # There is one thing we should fix: for some buy/sell statements the
417        # currency is incorrectly set to `self.currency`. It should be

421        #
422        # A better solution would be to fix this in the importer.
423        new_entries = self._fix_currency_for_buy_sell(new_entries)
424
425        # At this point, the transactions are well-formed. Now we will merge
426        # transactions with the same time & date.
427        old_entries = new_entries
428        new_entries = []
429        for entry in old_entries:
430            if (
431                len(new_entries) > 0
432                and new_entries[-1].date == entry.date
433                and new_entries[-1].meta["time"] == entry.meta["time"]
434            ):
435                for posting in entry.postings:
436                    existing_posting_pos = None
437                    for i in range(len(new_entries[-1].postings)):
438                        existing_posting = new_entries[-1].postings[i]
439                        if posting.account == existing_posting.account:
440                            if posting.units is None and existing_posting.units is None:
441                                existing_posting_pos = i
442                                break
443                            if (
444                                posting.units.currency == existing_posting.units.currency
445                                and posting.price == existing_posting.price
446                                and posting.cost == existing_posting.cost
447                            ):
448                                existing_posting_pos = i
449                                break
450
451                    if existing_posting_pos is None:
452                        new_entries[-1].postings.append(posting)
453                    # We can merge with an existing posting
454                    elif new_entries[-1].postings[existing_posting_pos].units is not None:
455                        new_entries[-1].postings[existing_posting_pos] = (
456                            new_entries[-1]
457                            .postings[existing_posting_pos]
458                            ._replace(
459                                units=new_entries[-1]
460                                .postings[existing_posting_pos]
461                                .units._replace(
462                                    number=new_entries[-1].postings[existing_posting_pos].units.number
463                                    + posting.units.number
464                                )
465                            )
466                        )
467            else:
468                new_entries.append(entry)
469
470        # We simplify transactions as follows: if there is a cost not
471        # in `self.currency`, we convert it by using the cost of this currency
472        # in terms of `self.currency`. We can then remove the adding and
473        # subtracting of this other currency.
474        old_entries = new_entries
475        new_entries = []
476        for entry in old_entries:
477            costs = [[posting.units.currency, posting.cost] for posting in entry.postings if posting.cost is not None]
478
479            for i in range(len(entry.postings)):
480                posting = entry.postings[i]
481                if posting.cost is not None and posting.cost.currency != self.currency:
482                    correct_posting = None
483                    for from_cur, cost in costs:
484                        if from_cur == posting.cost.currency:
485                            correct_posting = cost
486                    if correct_posting is None:
487                        continue
488                    entry.postings[i] = posting._replace(
489                        cost=posting.cost._replace(
490                            currency=self.currency,
491                            number=(posting.cost.number * correct_posting.number).quantize(decimal.Decimal("0.0001")),
492                        )
493                    )
494
495            new_entries.append(entry)
496
497        # We simplify one last time, to filter out combinations like 100 USD {0.8 EUR} & -100 USD {}
498        old_entries = new_entries
499        new_entries = []
500        for entry in old_entries:
501            cancelling_duo = None
502            for i in range(len(entry.postings)):
503                for j in range(i + 1, len(entry.postings)):
504                    if (
505                        entry.postings[i].account == entry.postings[j].account
506                        and entry.postings[i].units is not None
507                        and entry.postings[j].units is not None
508                        and entry.postings[i].units.currency == entry.postings[j].units.currency
509                        and entry.postings[i].units.number == -entry.postings[j].units.number
510                    ):
511                        cancelling_duo = [i, j]
512
513            if cancelling_duo is not None:
514                i, j = cancelling_duo
515                entry.postings.pop(j)
516                entry.postings.pop(i)
517
518            new_entries.append(entry)
